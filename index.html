// Google Apps Script for SyncSchool // This receives form submissions and writes them to your Google Sheet // IMPORTANT: Replace this with your actual Google Sheet ID const SHEET_ID = '1GYqRYExU0M8w7rg6PihNvWq_zt_EYriJ_0DiK868RKg'; const SHEET_NAME = 'Signups'; // You'll create this tab in your spreadsheet function doPost(e) { try { // Parse the incoming data const data = JSON.parse(e.postData.contents); // Open the spreadsheet const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); // Create the sheet if it doesn't exist if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); // Add headers sheet.appendRow([ 'Timestamp', 'Parent Name', 'Parent Email', 'Parent Phone', 'Child Name', 'Grade', 'Age Group', 'Activities', 'Extra Lessons', 'Notes' ]); // Format header row sheet.getRange(1, 1, 1, 10).setFontWeight('bold').setBackground('#f5f1e8'); } // Append the data sheet.appendRow([ data.timestamp, data.parentName, data.parentEmail, data.parentPhone, data.childName, data.grade, data.ageGroup, data.activities, data.extraLessons, data.notes ]); // Send success response return ContentService .createTextOutput(JSON.stringify({ status: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { // Send error response return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // Test function - run this to make sure everything works function test() { const testData = { timestamp: new Date().toISOString(), parentName: 'Test Parent', parentEmail: 'test@example.com', parentPhone: '123456789', childName: 'Test Child', grade: 'Grade 5', ageGroup: 'U11', activities: 'Hockey, Tennis', extraLessons: 'Monday - isiXhosa, Tuesday - Maths', notes: 'Test notes' }; doPost({ postData: { contents: JSON.stringify(testData) } }); Logger.log('Test completed - check your spreadsheet!'); }
